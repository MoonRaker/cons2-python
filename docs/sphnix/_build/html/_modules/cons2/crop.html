    def get_nckc(self):
        """
        Reads in crop coefficients.

        Parameters
        ----------
        name: string
            Name of the crop

        Returns
        -------
        nckc: list
            List of crop coefficients
        """

        try:
            infile = open(os.path.join(self.directory,'data','crop_coef_nckc.csv'),'r')
        except TypeError:
            logger.critical('crop_coef_nckc.csv file not found.')
            raise
        lines = infile.readlines()
        infile.close()

        nckca = [float(item) for item in lines[0].split(',')[1:]]
        nckcp = [float(item) for item in lines[1].split(',')[1:]]

        if self.crop_type == 'PERENNIAL':
            self.nckc= nckcp
        else:
            self.nckc = nckca


    def get_ckc(self):
